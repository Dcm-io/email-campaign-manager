        function navigateToSection(sectionId, stepId) {
            // Hide all sections
            document.querySelectorAll('.section').forEach(section => {
                section.classList.remove('active');
            });
            
            // Show target section
            document.getElementById(sectionId).classList.add('active');
            
            // Update active step
            document.querySelectorAll('.step').forEach(step => {
                step.classList.remove('active');
            });
            document.getElementById(stepId).classList.add('active');
        }
        
        // Template preview functionality
        document.getElementById('updatePreviewBtn').addEventListener('click', function() {
            const name = document.getElementById('previewName').value || 'John Doe';
            const content = document.getElementById('emailContent').value || 
                '<p>Hello {name},</p><p>This is a sample email template.</p><p>Regards,<br>Your Team</p>';
            
            const previewContent = content.replace(/{name}/g, name);
            document.getElementById('templatePreview').innerHTML = previewContent;
        });

            if (!file) return;
            
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const data = new Uint8Array(e.target.result);
                    const workbook = XLSX.read(data, { type: 'array' });
                    const firstSheetName = workbook.SheetNames[0];
                    const worksheet = workbook.Sheets[firstSheetName];
                    
                    // Convert to JSON
                    const jsonData = XLSX.utils.sheet_to_json(worksheet);
                    
                    // Update preview table
                    const tableBody = document.getElementById('contactListPreview');
                    tableBody.innerHTML = '';
                    
                    jsonData.slice(0, 5).forEach((row, index) => {
                        const tr = document.createElement('tr');
                        tr.innerHTML = `
                            <td>${index + 1}</td>
                            <td>${row.Email || row.email || ''}</td>
                            <td>${row.Name || row.name || ''}</td>
                            <td>${row.Subject || row.subject || ''}</td>
                            <td><span class="badge-status badge-pending">Pending</span></td>
                        `;
                        tableBody.appendChild(tr);
                    });
                    
                    if (jsonData.length > 5) {
                        const tr = document.createElement('tr');
                        tr.innerHTML = `
                            <td colspan="5" class="text-center text-muted">
                                + ${jsonData.length - 5} more contacts
                            </td>
                        `;
                        tableBody.appendChild(tr);
                    }
                    
                    // Update stats
                    document.getElementById('totalContacts').textContent = jsonData.length;
                    document.getElementById('sentEmails').textContent = '0';
                    document.getElementById('remainingEmails').textContent = jsonData.length;
                    document.getElementById('successRate').textContent = '0%';
                    
                    // Update campaign info
                    document.getElementById('campaignSubject').textContent = 
                        document.getElementById('emailSubject').value || 'Not set';
                    document.getElementById('campaignTimeGap').textContent = 
                        document.getElementById('timeGap').value;
                    document.getElementById('campaignDailyLimit').textContent = 
                        document.getElementById('dailySendingLimit').value;
                    
                    // Convert time to 12-hour format
                    const timeValue = document.getElementById('startTime').value;
                    const [hours, minutes] = timeValue.split(':');
                    let hours12 = hours % 12 || 12;
                    const ampm = hours >= 12 ? 'PM' : 'AM';
                    document.getElementById('campaignStartTime').textContent = 
                        `${hours12}:${minutes} ${ampm}`;
                    
                    // Add log entry
                    addLogEntry(`Successfully uploaded ${jsonData.length} contacts`);
                    
                } catch (error) {
                    addLogEntry(`Error importing contacts: ${error.message}`);
                }
            };
            reader.readAsArrayBuffer(file);
        });

        // Campaign controls
        let campaignActive = false;
        let sentCount = 0;
        let successCount = 0;
        let totalContacts = 0;
        
        document.getElementById('startCampaignBtn').addEventListener('click', function() {
            totalContacts = parseInt(document.getElementById('totalContacts').textContent) || 0;
            if (totalContacts === 0) {
                addLogEntry('Error: Please upload contacts before starting campaign');
                return;
            }
            
            campaignActive = true;
            this.disabled = true;
            document.getElementById('pauseCampaignBtn').disabled = false;
            
            addLogEntry('Campaign started! Emails will be sent with rotation between accounts.');
            
            // Start processing emails
            processNextEmail();
        });
        
        document.getElementById('pauseCampaignBtn').addEventListener('click', function() {
            campaignActive = false;
            this.disabled = true;
            document.getElementById('startCampaignBtn').disabled = false;
            
            addLogEntry('Campaign paused');
        });
        
        document.getElementById('resetCampaignBtn').addEventListener('click', function() {
            campaignActive = false;
            sentCount = 0;
            successCount = 0;
            
            document.getElementById('sentEmails').textContent = '0';
            document.getElementById('remainingEmails').textContent = totalContacts;
            document.getElementById('successRate').textContent = '0%';
            document.getElementById('progressBar').style.width = '0%';
            document.getElementById('progressPercent').textContent = '0%';
            
            document.getElementById('startCampaignBtn').disabled = false;
            document.getElementById('pauseCampaignBtn').disabled = true;
            
            addLogEntry('Campaign has been reset');
        });
        
        function processNextEmail() {
            if (!campaignActive || sentCount >= totalContacts) {
                // Campaign completed
                campaignActive = false;
                document.getElementById('startCampaignBtn').disabled = false;
                document.getElementById('pauseCampaignBtn').disabled = true;
                addLogEntry('Campaign completed!');
                return;
            }
            
            const minDelay = parseInt(document.getElementById('timeGap').value) * 60000 || 240000;
            const maxDelay = minDelay + 60000; // Add 1 minute for variation
            
            // Determine which sender to use (alternating between accounts)
            const sender = sentCount % 2 === 0 ? 'les@prosvio.com' : 'amy.market@growth-puls.com';
            
            // Simulate sending delay
            setTimeout(() => {
                // Simulate sending success (90% success rate)
                const success = Math.random() > 0.1;
                
                if (success) {
                    successCount++;
                }
                
                sentCount++;
                
                // Update stats
                document.getElementById('sentEmails').textContent = sentCount;
                document.getElementById('remainingEmails').textContent = totalContacts - sentCount;
                
                const successRate = Math.round((successCount / sentCount) * 100);
                document.getElementById('successRate').textContent = `${successRate}%`;
                
                const progress = Math.round((sentCount / totalContacts) * 100);
                document.getElementById('progressBar').style.width = `${progress}%`;
                document.getElementById('progressPercent').textContent = `${progress}%`;
                
                // Log the activity
                const action = success ? 'sent' : 'failed to send';
                addLogEntry(`${sender} ${action} email ${sentCount} of ${totalContacts}`);
                
                // Process next email
                processNextEmail();
            }, Math.floor(Math.random() * (maxDelay - minDelay + 1)) + minDelay);
        }
        
        // Add log entry
        function addLogEntry(message) {
            const now = new Date();
            const timestamp = now.toLocaleTimeString([], { hour: '2-digit', minute:'2-digit' });
            
            const logEntry = document.createElement('div');
            logEntry.className = 'log-entry';
            logEntry.innerHTML = `
                <div class="log-time">${timestamp}:</div>
                <div>${message}</div>
            `;
            
            document.getElementById('logContainer').prepend(logEntry);
            
            // Keep log to 50 entries max
            if (document.getElementById('logContainer').children.length > 50) {
                document.getElementById('logContainer').removeChild(
                    document.getElementById('logContainer').lastChild
                );
            }
        }
